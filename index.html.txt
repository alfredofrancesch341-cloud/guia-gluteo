    <script>
        function showTab(tabId, btn) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            document.querySelectorAll('.nav-btn').forEach(b => {
                b.classList.remove('active');
                b.classList.add('text-slate-500');
            });
            btn.classList.add('active');
            btn.classList.remove('text-slate-500');
        }

        function updateNutrition() {
            const weight = parseFloat(document.getElementById('weightInput').value);
            document.getElementById('weightVal').innerText = weight + ' kg';
            const kcal = Math.round(weight * 48); 
            const prot = Math.round(weight * 2.1);
            document.getElementById('kcalDisplay').innerText = kcal;
            document.getElementById('protDisplay').innerText = prot + 'g';
        }

        updateNutrition();
    </script>
